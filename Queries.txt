-- Queries 1
select
	T.Name,
	CSS.Year,
	CSS.Wins
from
	COACH_SEASON_STAT as CSS,
	TEAM_YEAR as TY
join
	TEAM as T on T.ID = TY.Team_ID
where
	CSS.Year = TY.Year
	and CSS.Coach_ID = TY.Coach_ID
	and TY.Year > 2021
	and TY.Team_ID = 2;

-- Queries 2
select
	P.Name,
	PY.Year,
	T.Name
from
	PLAYER_YEAR as PY
join
	TEAM as T on PY.Team_ID = T.ID,
	PLAYER as P on PY.Player_ID = P.ID
where
	P.ID = 3;

-- Queries 3
select
	C.ID,
	C.Name,
	CSS.Wins
from
	PLAYER as P
join
	PLAYER_YEAR as PY on P.ID = PY.Team_ID,
	TEAM_YEAR as TY on PY.Team_ID = TY.Team_ID and PY.Year = TY.Year,
	COACH_SEASON_STAT as CSS on TY.Coach_ID = CSS.Coach_ID and TY.Year = CSS.Year,
	COACH as C on C.ID = CSS.Coach_ID
where
	P.ID = 3
	and CSS.Wins < 10;

-- Queries 4
select
	T.Name,
	PY.Year,
	count(PY.Player_ID)
from
	PLAYER_YEAR as PY
join
	TEAM as T on PY.Team_ID = T.ID
where
	PY.Team_ID = 2
group by
	PY.Year;

-- Queries 5
select
	SP.TeamID,
	SP.TeamName,
	CSS.Coach_ID,
	CSS.Wins,
	CSS.Losses,
	CSS.Num_Passes,
	CSS.Num_Runs,
	SP.Year,
	max(SP.SumPoint)
from
	(
		select
			T.ID as TeamID,
			T.Name as TeamName,
			G.Year as Year,
			SUM(case when G.Home_Team_ID = T.ID then G.Home_Points else G.Away_Points end) as SumPoint
		from
			TEAM as T,
			GAME as G
		where
			G.Home_Team_ID = T.ID
			or G.Away_Team_ID = T.ID
		group by
			T.ID,
			G.Year
	) as SP
join
	TEAM_YEAR as TY on SP.Year = TY.Year and TY.Team_ID = SP.TeamID,
	COACH_SEASON_STAT as CSS on TY.Coach_ID = CSS.Coach_ID and TY.Year = CSS.Year
group by
	SP.TeamID;

-- Queries 6
select
	C.Name,
	C.ID,
	CMAX.SW
from
	(select
		Coach_ID,
		SUM(Wins) as SW
	from
		COACH_SEASON_STAT as CSS
	group by
		Coach_ID
	order by
		SW
	limit 1) as CMAX,
	COACH as C
where
	C.ID = CMAX.Coach_ID;
